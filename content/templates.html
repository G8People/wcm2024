            background-color: white;

        #message {
            text-align: center;
            font-size: 24px;

<body>
    <table id="board"></table>
    <div id="message">点击鼠标左键开始游戏</div>
    <script>
        const boardSize = 15;
        let currentPlayer = 'X';
        let board = Array.from({ length: boardSize }, () => Array(boardSize).fill(''));
        let gameStarted = false;

        function renderBoard() {
            const boardElem = document.getElementById('board');
            boardElem.innerHTML = '';
            for (let i = 0; i < boardSize; i++) {
                const row = document.createElement('tr');
                for (let j = 0; j < boardSize; j++) {
                    const cell = document.createElement('td');
                    if (board[i][j] !== '') {
                        const piece = document.createElement('div');
                        piece.className = `piece ${board[i][j]}`;
                        cell.appendChild(piece);
                    }
                    cell.addEventListener('click', () => makeMove(i, j));
                    row.appendChild(cell);
                }
                boardElem.appendChild(row);
            }
        }

        function makeMove(x, y) {
            if (!gameStarted) return;
            fetch('/move', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify({ x, y })
            })
            .then(response => response.json())
            .then(data => {
                if (data.error) {
                    alert(data.error);
                } else {
                    board = data.board;
                    renderBoard();
                    if (data.winner) {
                        document.getElementById('message').textContent = `${data.winner} wins! 游戏结束，按下空白键重新开始`;
                        gameStarted = false;
                        document.addEventListener('keydown', handleRestart);
                    }
                }
            });
        }

        function startGame() {
            gameStarted = true;
            document.getElementById('message').textContent = '游戏已开始';
            document.removeEventListener('click', startGame);
        }

        function resetGame() {
            fetch('/reset', {
                method: 'POST'
            })
            .then(response => response.json())
            .then(data => {
                board = data.board;
                renderBoard();
                document.getElementById('message').textContent = '点击鼠标左键开始游戏';
                document.addEventListener('click', startGame);
            });
        }

        function handleRestart(event) {
            if (event.code === 'Space') {
                resetGame();
                document.removeEventListener('keydown', handleRestart);
            }
        }

        document.addEventListener('click', startGame);

        renderBoard();
    </script>
</body>
</html>
